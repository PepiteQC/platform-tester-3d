'use client';

import { useEffect, useState } from 'react';
import { Card } from '@/components/ui/card';
import { Brain, TrendingUp, Zap } from 'lucide-react';

interface TROXTInsight {
  bestPerformingStyle: string;
  averageGenerationTime: number;
  successRate: number;
  recommendations: string[];
}

export function TROXTInsightsPanel() {
  const [insights, setInsights] = useState<TROXTInsight | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetchInsights();
  }, []);

  const fetchInsights = async () => {
    try {
      const response = await fetch('/api/troxt/insights');
      const data = await response.json();
      if (data.success && data.insights) {
        setInsights(data.insights);
      }
    } catch (error) {
      console.error('[v0] Failed to fetch TROXT insights:', error);
    } finally {
      setLoading(false);
    }
  };

  if (loading || !insights) {
    return null;
  }

  return (
    <Card className="p-4 bg-gradient-to-br from-primary/10 to-accent/10 border-primary/30">
      <div className="flex items-start gap-3 mb-3">
        <Brain className="w-5 h-5 text-primary mt-0.5 flex-shrink-0" />
        <div className="flex-1">
          <h3 className="font-semibold text-foreground text-sm mb-2">TROXT Intelligence</h3>
          <div className="space-y-2">
            <div className="flex items-center justify-between text-xs">
              <span className="text-foreground/60">Best Style:</span>
              <span className="font-medium text-primary capitalize">{insights.bestPerformingStyle}</span>
            </div>
            <div className="flex items-center justify-between text-xs">
              <span className="text-foreground/60">Avg Time:</span>
              <span className="font-medium text-primary">{insights.averageGenerationTime}s</span>
            </div>
            <div className="flex items-center justify-between text-xs">
              <span className="text-foreground/60">Success:</span>
              <span className="font-medium text-primary">{insights.successRate}%</span>
            </div>
          </div>
        </div>
      </div>

      {insights.recommendations.length > 0 && (
        <div className="pt-3 border-t border-border/30">
          <p className="text-xs text-foreground/60 mb-2 flex items-center gap-1">
            <TrendingUp className="w-3 h-3" />
            Recommendations
          </p>
          <p className="text-xs text-foreground/80 leading-relaxed">
            {insights.recommendations[0]}
          </p>
        </div>
      )}
    </Card>
  );
}
