'use client';

import type React from 'react';
import { useCallback, useRef, useState, useEffect } from 'react';
import dynamic from 'next/dynamic';
import { Tabs, TabsContent, TabsList, TabsTrigger } from '@/components/ui/tabs';
import { Textarea } from '@/components/ui/textarea';
import { Button } from '@/components/ui/button';
import { Label } from '@/components/ui/label';
import {
  Boxes,
  Download,
  ImagePlus,
  Loader2,
  Sparkles,
  Type,
  Upload,
  X,
  RotateCw,
  Brain,
  Zap,
  TrendingUp,
} from 'lucide-react';
import { toast } from 'sonner';
import { cn } from '@/lib/utils';
import { TROXTInsightsPanel } from './troxt-insights-panel';

const ModelViewer = dynamic(
  () => import('@/components/model-viewer').then((m) => m.ModelViewer),
  { ssr: false }
);

type Phase = 'idle' | 'troxt-analyzing' | 'submitting' | 'processing' | 'done' | 'error';

const QUALITY_OPTIONS = [
  { value: 'high', label: 'High', hint: 'Maximum detail' },
  { value: 'balanced', label: 'Balanced', hint: 'Recommended' },
  { value: 'fast', label: 'Fast', hint: 'Quick preview' },
] as const;

interface ResultFile {
  name: string;
  url: string;
}

interface TROXTAnalysis {
  finalPrompt: string;
  strategy: {
    quality: string;
    estimatedTime: number;
  };
  promptAnalysis?: {
    parsed: {
      complexity: string;
      style: string;
    };
  };
}

const POLL_INTERVAL = 4000;
const MAX_POLLS = 120;

export function GeneratorWithTROXT() {
  const [mode, setMode] = useState<'text' | 'image'>('text');
  const [prompt, setPrompt] = useState('');
  const [quality, setQuality] = useState<string>('balanced');
  const [images, setImages] = useState<File[]>([]);
  const [previews, setPreviews] = useState<string[]>([]);
  const [phase, setPhase] = useState<Phase>('idle');
  const [statusText, setStatusText] = useState('');
  const [modelUrl, setModelUrl] = useState<string | null>(null);
  const [files, setFiles] = useState<ResultFile[]>([]);
  const [dragActive, setDragActive] = useState(false);
  const [troxtAnalysis, setTroxtAnalysis] = useState<TROXTAnalysis | null>(null);
  const [showTroxtDetails, setShowTroxtDetails] = useState(false);
  const fileInputRef = useRef<HTMLInputElement>(null);
  const generationIdRef = useRef<string>('');

  const busy = phase === 'troxt-analyzing' || phase === 'submitting' || phase === 'processing';

  const addFiles = useCallback((incoming: FileList | File[]) => {
    const accepted = Array.from(incoming).filter((f) => f.type.startsWith('image/'));
    if (accepted.length === 0) return;
    setImages((prev) => [...prev, ...accepted].slice(0, 5));
    setPreviews((prev) =>
      [...prev, ...accepted.map((f) => URL.createObjectURL(f))].slice(0, 5)
    );
  }, []);

  const removeImage = useCallback((index: number) => {
    setImages((prev) => prev.filter((_, i) => i !== index));
    setPreviews((prev) => prev.filter((_, i) => i !== index));
  }, []);

  const reset = useCallback(() => {
    setModelUrl(null);
    setFiles([]);
    setPhase('idle');
    setStatusText('');
    setTroxtAnalysis(null);
  }, []);

  // Analyze with TROXT before generation
  const analyzeTROXT = useCallback(async () => {
    if (mode === 'text' && !prompt.trim()) {
      toast.error('Enter a prompt for TROXT to analyze.');
      return false;
    }
    if (mode === 'image' && images.length === 0) {
      toast.error('Upload at least one image for TROXT to analyze.');
      return false;
    }

    setPhase('troxt-analyzing');
    setStatusText('TROXT analyzing your input...');

    try {
      if (mode === 'text') {
        const response = await fetch('/api/troxt/analyze-prompt', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ prompt: prompt.trim() }),
        });

        if (!response.ok) throw new Error('TROXT analysis failed');
        const data = await response.json();

        if (!data.success) throw new Error(data.error);

        setTroxtAnalysis(data.analysis);
        setQuality(data.analysis.strategy.quality);
        toast.success('TROXT analysis complete');
        return true;
      } else {
        // Image analysis
        const formData = new FormData();
        formData.append('file', images[0]);
        formData.append('prompt', prompt || '');

        const response = await fetch('/api/troxt/analyze-image', {
          method: 'POST',
          body: formData,
        });

        if (!response.ok) throw new Error('TROXT image analysis failed');
        const data = await response.json();

        if (!data.success) throw new Error(data.error);

        setTroxtAnalysis(data.analysis);
        setQuality(data.analysis.strategy.quality);
        toast.success('TROXT image analysis complete');
        return true;
      }
    } catch (error) {
      console.error('[v0] TROXT analysis error:', error);
      toast.error(`TROXT analysis failed: ${String(error)}`);
      setPhase('error');
      return false;
    }
  }, [mode, prompt, images]);

  const generate = useCallback(
    async () => {
      // First analyze with TROXT
      const analyzed = await analyzeTROXT();
      if (!analyzed) return;

      setPhase('submitting');
      setStatusText('Submitting to Rodin API…');

      try {
        const response = await fetch('/api/generate', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            prompt: troxtAnalysis?.finalPrompt || prompt,
            mode,
            quality,
          }),
        });

        if (!response.ok) throw new Error('Generation submission failed');
        const data = await response.json();

        if (!data.success) throw new Error(data.error);

        generationIdRef.current = data.taskId;

        // Record generation with TROXT
        if (troxtAnalysis) {
          await fetch('/api/troxt/record-generation', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              input: mode === 'text' ? prompt : 'image',
              inputType: mode,
              strategy: troxtAnalysis.strategy,
              complexity: troxtAnalysis.promptAnalysis?.parsed.complexity || 'moderate',
              style: troxtAnalysis.promptAnalysis?.parsed.style || 'realistic',
              success: true,
              generationTime: 0,
              quality: 0,
            }),
          });
        }

        setPhase('processing');
        setStatusText(`Generating (estimated ${troxtAnalysis?.strategy.estimatedTime || 45}s)…`);

        // Poll for completion
        let pollCount = 0;
        while (pollCount < MAX_POLLS) {
          await new Promise((resolve) => setTimeout(resolve, POLL_INTERVAL));
          pollCount++;

          try {
            const statusResponse = await fetch(`/api/status?taskId=${data.taskId}`);
            if (!statusResponse.ok) continue;

            const statusData = await statusResponse.json();
            const progress = statusData.data?.progress || 0;
            setStatusText(
              `Generating (${progress}%)…`
            );

            if (statusData.data?.status === 'completed') {
              const downloadResponse = await fetch(
                `/api/download?taskId=${data.taskId}`
              );
              if (downloadResponse.ok) {
                const blob = await downloadResponse.blob();
                const url = URL.createObjectURL(blob);
                setModelUrl(url);
                setFiles([{ name: 'generated-model.glb', url }]);
              }

              setPhase('done');
              setStatusText('Generation complete');
              toast.success('Your 3D model is ready!');
              return;
            }
          } catch (e) {
            console.error('[v0] Status check error:', e);
          }
        }

        throw new Error('Generation timeout');
      } catch (error) {
        console.error('[v0] Generation error:', error);
        toast.error(`Generation failed: ${String(error)}`);
        setPhase('error');
      }
    },
    [prompt, quality, mode, troxtAnalysis, analyzeTROXT]
  );

  return (
    <div className="flex flex-col h-full gap-6">
      <div className="flex-1 min-h-0 flex flex-col lg:flex-row gap-6">
        {/* Left Panel - Input */}
        <div className="lg:w-96 flex flex-col gap-4 overflow-y-auto">
          {/* TROXT Insights */}
          <TROXTInsightsPanel />

          {/* Mode Tabs */}
          <Tabs
            value={mode}
            onValueChange={(v) => setMode(v as 'text' | 'image')}
            className="w-full"
          >
            <TabsList className="grid w-full grid-cols-2 bg-secondary/20">
              <TabsTrigger value="text" className="gap-2">
                <Type className="w-4 h-4" />
                Text
              </TabsTrigger>
              <TabsTrigger value="image" className="gap-2">
                <ImagePlus className="w-4 h-4" />
                Image
              </TabsTrigger>
            </TabsList>

            {/* Text Mode */}
            <TabsContent value="text" className="space-y-3 mt-4">
              <Label htmlFor="prompt" className="text-sm font-medium">
                Describe your 3D model
              </Label>
              <Textarea
                id="prompt"
                value={prompt}
                onChange={(e) => setPrompt(e.target.value)}
                placeholder="A detailed robot warrior, highly detailed, futuristic style, cinematic lighting..."
                className="min-h-32 resize-none bg-card/50"
                disabled={busy}
              />
            </TabsContent>

            {/* Image Mode */}
            <TabsContent value="image" className="space-y-3 mt-4">
              <Label className="text-sm font-medium">Upload images to analyze</Label>
              <div
                className={cn(
                  'border-2 border-dashed rounded-lg p-6 text-center transition-colors',
                  dragActive
                    ? 'border-primary bg-primary/10'
                    : 'border-border bg-card/20 hover:border-primary/50'
                )}
                onDragEnter={() => setDragActive(true)}
                onDragLeave={() => setDragActive(false)}
                onDrop={(e) => {
                  e.preventDefault();
                  setDragActive(false);
                  addFiles(e.dataTransfer.files);
                }}
              >
                <Upload className="w-8 h-8 mx-auto mb-2 text-primary" />
                <p className="text-sm font-medium mb-1">Drag images here</p>
                <p className="text-xs text-foreground/50 mb-3">or</p>
                <Button
                  size="sm"
                  variant="outline"
                  onClick={() => fileInputRef.current?.click()}
                  disabled={busy}
                >
                  Select Files
                </Button>
                <input
                  ref={fileInputRef}
                  type="file"
                  multiple
                  accept="image/*"
                  className="hidden"
                  onChange={(e) => e.target.files && addFiles(e.target.files)}
                />
              </div>

              {/* Image Previews */}
              {previews.length > 0 && (
                <div className="grid grid-cols-3 gap-2">
                  {previews.map((src, idx) => (
                    <div key={idx} className="relative aspect-square rounded-lg overflow-hidden bg-card/50 border border-border/30">
                      <img
                        src={src}
                        alt={`Preview ${idx}`}
                        className="w-full h-full object-cover"
                      />
                      <button
                        onClick={() => removeImage(idx)}
                        className="absolute top-1 right-1 p-1 rounded-full bg-destructive hover:bg-destructive/90"
                        disabled={busy}
                      >
                        <X className="w-3 h-3 text-white" />
                      </button>
                    </div>
                  ))}
                </div>
              )}

              {/* Additional Prompt for Images */}
              <Label htmlFor="image-prompt" className="text-sm font-medium">
                Additional description (optional)
              </Label>
              <Textarea
                id="image-prompt"
                value={prompt}
                onChange={(e) => setPrompt(e.target.value)}
                placeholder="Add style, mood, or other details..."
                className="min-h-20 resize-none bg-card/50"
                disabled={busy}
              />
            </TabsContent>
          </Tabs>

          {/* Quality Selector */}
          <div>
            <Label className="text-sm font-medium mb-2 block">Quality Preset</Label>
            <div className="grid grid-cols-3 gap-2">
              {QUALITY_OPTIONS.map((opt) => (
                <button
                  key={opt.value}
                  onClick={() => setQuality(opt.value)}
                  className={cn(
                    'p-3 rounded-lg border-2 transition-colors text-left',
                    quality === opt.value
                      ? 'border-primary bg-primary/10'
                      : 'border-border bg-card/30 hover:border-primary/50'
                  )}
                  disabled={busy}
                >
                  <div className="text-sm font-medium">{opt.label}</div>
                  <div className="text-xs text-foreground/50">{opt.hint}</div>
                </button>
              ))}
            </div>
          </div>

          {/* TROXT Analysis Details */}
          {troxtAnalysis && (
            <div className="p-3 rounded-lg border border-primary/30 bg-primary/5 space-y-2">
              <button
                onClick={() => setShowTroxtDetails(!showTroxtDetails)}
                className="flex items-center gap-2 text-xs font-medium text-primary w-full"
              >
                <Brain className="w-3 h-3" />
                TROXT Analysis
              </button>
              {showTroxtDetails && (
                <div className="text-xs space-y-1 pt-2 border-t border-primary/20">
                  <p className="text-foreground/70">
                    <span className="text-foreground/50">Complexity:</span>{' '}
                    {troxtAnalysis.promptAnalysis?.parsed.complexity}
                  </p>
                  <p className="text-foreground/70">
                    <span className="text-foreground/50">Style:</span>{' '}
                    {troxtAnalysis.promptAnalysis?.parsed.style}
                  </p>
                  <p className="text-foreground/70">
                    <span className="text-foreground/50">Est. Time:</span>{' '}
                    {troxtAnalysis.strategy.estimatedTime}s
                  </p>
                </div>
              )}
            </div>
          )}

          {/* Generate Button */}
          <Button
            onClick={generate}
            disabled={busy || (mode === 'text' && !prompt.trim()) || (mode === 'image' && images.length === 0)}
            className="w-full gap-2 bg-primary hover:bg-primary/90 text-primary-foreground"
            size="lg"
          >
            {busy ? (
              <>
                <Loader2 className="w-4 h-4 animate-spin" />
                {statusText}
              </>
            ) : (
              <>
                <Sparkles className="w-4 h-4" />
                Generate with TROXT
              </>
            )}
          </Button>

          {/* Reset Button */}
          {modelUrl && (
            <Button
              onClick={reset}
              variant="outline"
              className="w-full gap-2"
              size="sm"
            >
              <RotateCw className="w-4 h-4" />
              New Generation
            </Button>
          )}

          {/* Download Button */}
          {modelUrl && (
            <Button
              onClick={() => {
                const a = document.createElement('a');
                a.href = modelUrl;
                a.download = 'model.glb';
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
                toast.success('Model downloaded');
              }}
              variant="secondary"
              className="w-full gap-2"
              size="sm"
            >
              <Download className="w-4 h-4" />
              Download GLB
            </Button>
          )}
        </div>

        {/* Right Panel - Viewer */}
        <div className="flex-1 min-h-96 lg:min-h-auto">
          {modelUrl ? (
            <ModelViewer modelUrl={modelUrl} />
          ) : (
            <div className="w-full h-full rounded-lg border border-border/30 bg-card/20 flex items-center justify-center">
              <div className="text-center space-y-4">
                <Boxes className="w-16 h-16 mx-auto text-primary/40" />
                <div>
                  <p className="text-foreground/60 text-sm">Generate your first 3D model</p>
                  <p className="text-foreground/40 text-xs">Powered by TROXT Intelligence + Rodin</p>
                </div>
              </div>
            </div>
          )}
        </div>
      </div>
    </div>
  );
}
