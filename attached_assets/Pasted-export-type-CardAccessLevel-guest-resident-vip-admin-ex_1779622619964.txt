interface SecurityDoorProps {
  position?: [number, number, number]
  rotation?: [number, number, number]
  doorId: 'main' | 'bathroom'
  requiredLevel?: CardAccessLevel
}

export function SecurityDoor({ 
  position = [0, 0, 0], 
  rotation = [0, 0, 0],
  doorId,
  requiredLevel = 'resident'
}: SecurityDoorProps) {
  const { mainDoor, bathroomDoor, attemptDoorAccess, playerCard } = useEtherWorldStore()
  const door = doorId === 'main' ? mainDoor : bathroomDoor
  const doorRef = useRef<THREE.Group>(null)
  const [isAnimating, setIsAnimating] = useState(false)
  const targetRotation = useRef(0)
  
  // Door animation
  useFrame(() => {
    if (doorRef.current) {
      const target = door.isOpen ? -Math.PI / 2 : 0
      targetRotation.current = target
      
      const currentRotation = doorRef.current.rotation.y
      const diff = target - currentRotation
      
      if (Math.abs(diff) > 0.01) {
        doorRef.current.rotation.y += diff * 0.1
        setIsAnimating(true)
      } else {
        doorRef.current.rotation.y = target
        setIsAnimating(false)
      }
    }
  })
  
  const handleDoorClick = () => {
    if (!isAnimating) {
      attemptDoorAccess(doorId, playerCard)
    }
  }
  
  return (
    <group position={position} rotation={rotation}>
      {/* Door frame */}
      <DoorFrame />
      
      {/* Door panel (hinged on left side) */}
      <group ref={doorRef} position={[-0.45, 0, 0]}>
        <mesh 
          position={[0.45, 1.1, 0]} 
          castShadow 
          onClick={handleDoorClick}
        >
          <boxGeometry args={[0.9, 2.2, 0.08]} />
          <meshStandardMaterial 
            color="#1a1a2e" 
            metalness={0.4} 
            roughness={0.6} 
          />
        </mesh>
        
        {/* Door handle */}
        <mesh position={[0.8, 1.05, 0.06]}>
          <boxGeometry args={[0.12, 0.04, 0.04]} />
          <meshStandardMaterial color="#6b7280" metalness={0.9} roughness={0.1} />
        </mesh>
        <mesh position={[0.82, 1.05, 0.1]} rotation={[Math.PI / 2, 0, 0]}>
          <cylinderGeometry args={[0.02, 0.02, 0.08, 8]} />
          <meshStandardMaterial color="#6b7280" metalness={0.9} roughness={0.1} />
        </mesh>
        
        {/* Room number plate */}
        {doorId === 'main' && (
          <group position={[0.45, 1.8, 0.05]}>
            <mesh>
              <boxGeometry args={[0.2, 0.1, 0.01]} />
              <meshStandardMaterial color="#374151" metalness={0.7} roughness={0.3} />
            </mesh>
          </group>
        )}
        
        {/* Peephole */}
        {doorId === 'main' && (
          <mesh position={[0.45, 1.5, 0.045]} rotation={[Math.PI / 2, 0, 0]}>
            <cylinderGeometry args={[0.015, 0.015, 0.1, 8]} />
            <meshStandardMaterial color="#0a0a0a" metalness={0.5} roughness={0.5} />
          </mesh>
        )}
      </group>
      
      {/* Card reader */}
      {door.isLocked && (
        <CardReader 
          position={[0.6, 1.1, 0.05]} 
          doorId={doorId}
          requiredLevel={requiredLevel}
        />
      )}
    </group>
  )
}

interface CardReaderProps {
  position: [number, number, number]
  doorId: 'main' | 'bathroom'
  requiredLevel: CardAccessLevel
}

function CardReader({ position, doorId, requiredLevel }: CardReaderProps) {
  const { mainDoor, bathroomDoor, attemptDoorAccess, playerCard } = useEtherWorldStore()
  const door = doorId === 'main' ? mainDoor : bathroomDoor
  const [showPrompt, setShowPrompt] = useState(false)
  const [lastScanTime, setLastScanTime] = useState(0)
  const ledRef = useRef<THREE.Mesh>(null)
  const scanLineRef = useRef<THREE.Mesh>(null)
  
  // LED and scan animation
  useFrame((state) => {
    if (ledRef.current) {
      const material = ledRef.current.material as THREE.MeshStandardMaterial
      
      if (door.accessDenied) {
        // Red blinking for denied
        material.color.setHex(0xef4444)
        material.emissive.setHex(0xef4444)
        material.emissiveIntensity = 0.5 + Math.sin(state.clock.elapsedTime * 20) * 0.5
      } else if (door.isOpen) {
        // Green for open
        material.color.setHex(0x22c55e)
        material.emissive.setHex(0x22c55e)
        material.emissiveIntensity = 0.8
      } else {
        // Amber idle pulse
        material.color.setHex(0xf59e0b)
        material.emissive.setHex(0xf59e0b)
        material.emissiveIntensity = 0.3 + Math.sin(state.clock.elapsedTime * 2) * 0.2
      }
    }
    
    // Scan line animation
    if (scanLineRef.current && Date.now() - lastScanTime < 1000) {
      const progress = (Date.now() - lastScanTime) / 1000
      scanLineRef.current.position.y = 0.04 - progress * 0.08
      scanLineRef.current.visible = true
    } else if (scanLineRef.current) {
      scanLineRef.current.visible = false
    }
  })
  
  const handleScan = () => {
    setLastScanTime(Date.now())
    setTimeout(() => {
      attemptDoorAccess(doorId, playerCard)
    }, 500)
  }
  
  return (
    <group position={position}>
      {/* Reader body */}
      <mesh castShadow>
        <boxGeometry args={[0.1, 0.15, 0.03]} />
        <meshStandardMaterial color="#1a1a1a" metalness={0.5} roughness={0.5} />
      </mesh>
      
      {/* Card scan area */}
      <mesh 
        position={[0, -0.02, 0.016]}
        onClick={handleScan}
        onPointerEnter={() => setShowPrompt(true)}
        onPointerLeave={() => setShowPrompt(false)}
      >
        <boxGeometry args={[0.07, 0.08, 0.005]} />
        <meshStandardMaterial color="#0f172a" metalness={0.8} roughness={0.2} />
      </mesh>
      
      {/* Scan line */}
      <mesh ref={scanLineRef} position={[0, 0, 0.02]} visible={false}>
        <boxGeometry args={[0.06, 0.005, 0.001]} />
        <meshStandardMaterial color="#3b82f6" emissive="#3b82f6" emissiveIntensity={2} />
      </mesh>
      
      {/* LED indicator */}
      <mesh ref={ledRef} position={[0, 0.055, 0.016]}>
        <boxGeometry args={[0.03, 0.015, 0.005]} />
        <meshStandardMaterial color="#f59e0b" emissive="#f59e0b" emissiveIntensity={0.5} />
      </mesh>
      
      {/* Required level indicator */}
      <mesh position={[0, -0.065, 0.016]}>
        <boxGeometry args={[0.04, 0.01, 0.002]} />
        <meshStandardMaterial 
          color={CARD_COLORS[requiredLevel]} 
          emissive={CARD_COLORS[requiredLevel]} 
          emissiveIntensity={0.3} 
        />
      </mesh>
      
      {/* Prompt tooltip */}
      {showPrompt && (
        <Html position={[0.15, 0, 0]} center>
          <div className="bg-background/95 border border-border px-3 py-2 rounded-lg shadow-xl whitespace-nowrap">
            <p className="text-xs text-foreground font-medium">Scanner la carte</p>
            <p className="text-xs text-muted-foreground">Niveau requis: {requiredLevel}</p>
          </div>
        </Html>
      )}
    </group>
  )
}

'use client'

import { useState, useCallback } from 'react'
import { useEtherWorldStore } from '@/lib/etherworld/store'
import { RARITY_COLORS, CATEGORY_ICONS, type InventoryItem, type ItemCategory, type ItemRarity } from '@/lib/etherworld/types'
import { cn } from '@/lib/utils'
import {
  X,
  Package,
  Trash2,
  Weight,
  Grid3x3,
} from 'lucide-react'

type SortMode = 'name' | 'rarity' | 'category' | 'weight' | 'value'

const RARITY_ORDER: Record<ItemRarity, number> = {
  common: 0,
  uncommon: 1,
  rare: 2,
  epic: 3,
  legendary: 4,
}

export function InventoryPanel() {
  const {
    inventoryOpen,
    inventory,
    inventoryConfig,
    selectedSlot,
    toggleInventory,
    removeItem,
    useItem,
    moveItem,
    selectSlot,
    playerStats,
  } = useEtherWorldStore()

  const [sortMode, setSortMode] = useState<SortMode>('name')
  const [filterCategory, setFilterCategory] = useState<ItemCategory | 'all'>('all')
  const [dragFrom, setDragFrom] = useState<number | null>(null)

  const totalWeight = inventory.reduce((sum, s) => sum + (s.item ? s.item.weight * s.item.quantity : 0), 0)

  const sortedInventory = [...inventory].sort((a, b) => {
    if (!a.item && !b.item) return 0
    if (!a.item) return 1
    if (!b.item) return -1
    switch (sortMode) {
      case 'rarity':
        return RARITY_ORDER[b.item.rarity] - RARITY_ORDER[a.item.rarity]
      case 'category':
        return a.item.category.localeCompare(b.item.category)
      case 'weight':
        return b.item.weight - a.item.weight
      case 'value':
        return b.item.value - a.item.value
      default:
        return a.item.name.localeCompare(b.item.name)
    }
  })

  const filteredInventory = filterCategory === 'all'
    ? sortedInventory
    : sortedInventory.filter((s) => s.item?.category === filterCategory)

  const handleDrop = useCallback((toSlot: number) => {
    if (dragFrom !== null && dragFrom !== toSlot) {
      moveItem(dragFrom, toSlot)
    }
    setDragFrom(null)
  }, [dragFrom, moveItem])

  if (!inventoryOpen) return null

  return (
    <div className="fixed inset-0 z-[160] flex items-center justify-center pointer-events-none">
      <div className="pointer-events-auto" onClick={(e) => e.stopPropagation()}>
        <div className="bg-zinc-900/95 backdrop-blur-md border border-zinc-700/50 rounded-xl shadow-2xl w-[640px] max-h-[80vh] flex flex-col overflow-hidden">
          {/* Header */}
          <div className="flex items-center justify-between px-4 py-3 border-b border-zinc-700/50">
            <div className="flex items-center gap-2">
              <Package className="w-4 h-4 text-amber-400" />
              <span className="text-sm font-bold text-white tracking-wider">INVENTAIRE</span>
            </div>
            <div className="flex items-center gap-3">
              <div className="flex items-center gap-1 text-[10px] text-zinc-400">
                <Weight className="w-3 h-3" />
                <span>{totalWeight.toFixed(1)}/{inventoryConfig.maxWeight}kg</span>
              </div>
              <div className="flex items-center gap-1 text-[10px] text-zinc-400">
                <Grid3x3 className="w-3 h-3" />
                <span>{inventory.filter((s) => s.item).length}/{inventoryConfig.maxSlots}</span>
              </div>
              <button onClick={toggleInventory} className="p-1 hover:bg-zinc-700/50 rounded transition-colors">
                <X className="w-4 h-4 text-zinc-400" />
              </button>
            </div>
          </div>

          {/* Toolbar */}
          <div className="flex items-center gap-2 px-4 py-2 border-b border-zinc-700/30">
            <div className="flex items-center gap-1">
              <span className="text-[10px] text-zinc-500 uppercase tracking-wider">Tri:</span>
              {(['name', 'rarity', 'category', 'weight', 'value'] as SortMode[]).map((mode) => (
                <button
                  key={mode}
                  onClick={() => setSortMode(mode)}
                  className={cn(
                    "px-2 py-0.5 rounded text-[10px] font-semibold transition-all",
                    sortMode === mode
                      ? "bg-amber-500/20 text-amber-400"
                      : "text-zinc-500 hover:text-zinc-300"
                  )}
                >
                  {mode === 'name' ? 'Nom' : mode === 'rarity' ? 'Rareté' : mode === 'category' ? 'Type' : mode === 'weight' ? 'Poids' : 'Valeur'}
                </button>
              ))}
            </div>
            <div className="ml-auto flex items-center gap-1">
              <span className="text-[10px] text-zinc-500 uppercase tracking-wider">Filtre:</span>
              <select
                value={filterCategory}
                onChange={(e) => setFilterCategory(e.target.value as ItemCategory | 'all')}
                className="bg-zinc-800 border border-zinc-700 rounded px-1.5 py-0.5 text-[10px] text-zinc-300"
              >
                <option value="all">Tous</option>
                {Object.entries(CATEGORY_ICONS).map(([cat, icon]) => (
                  <option key={cat} value={cat}>{icon} {cat}</option>
                ))}
              </select>
            </div>
          </div>

          {/* Grid */}
          <div className="flex-1 overflow-y-auto p-4">
            <div className="grid grid-cols-8 gap-1.5">
              {filteredInventory.map((slot) => (
                <InventorySlotComponent
                  key={slot.slotId}
                  slot={slot}
                  isSelected={selectedSlot === slot.slotId}
                  onSelect={() => selectSlot(slot.slotId)}
                  onUse={() => useItem(slot.slotId)}
                  onRemove={() => removeItem(slot.slotId)}
                  onDragStart={() => setDragFrom(slot.slotId)}
                  onDrop={() => handleDrop(slot.slotId)}
                />
              ))}
            </div>
          </div>

          {/* Selected item detail */}
          {selectedSlot !== null && (() => {
            const slot = inventory.find((s) => s.slotId === selectedSlot)
            if (!slot?.item) return null
            const item = slot.item
            return (
              <div className="border-t border-zinc-700/30 px-4 py-3">
                <div className="flex items-start gap-3">
                  <div
                    className="w-12 h-12 rounded-lg flex items-center justify-center text-2xl border"
                    style={{
                      borderColor: RARITY_COLORS[item.rarity],
                      backgroundColor: `${RARITY_COLORS[item.rarity]}15`,
                    }}
                  >
                    {item.icon}
                  </div>
                  <div className="flex-1 min-w-0">
                    <div className="flex items-center gap-2">
                      <span className="text-sm font-bold text-white">{item.name}</span>
                      <span
                        className="text-[10px] font-semibold uppercase px-1.5 py-0.5 rounded"
                        style={{
                          color: RARITY_COLORS[item.rarity],
                          backgroundColor: `${RARITY_COLORS[item.rarity]}20`,
                        }}
                      >
                        {item.rarity}
                      </span>
                    </div>
                    <p className="text-xs text-zinc-400 mt-0.5">{item.description}</p>
                    <div className="flex items-center gap-3 mt-1.5 text-[10px] text-zinc-500">
                      <span>{CATEGORY_ICONS[item.category]} {item.category}</span>
                      <span>Poids: {item.weight}kg</span>
                      <span>Valeur: {item.value}cr</span>
                      {item.stackable && <span>x{item.quantity}/{item.maxStack}</span>}
                    </div>
                  </div>
                  <div className="flex flex-col gap-1">
                    {item.usable && (
                      <button
                        onClick={() => useItem(slot.slotId)}
                        className="px-3 py-1 bg-emerald-500/20 hover:bg-emerald-500/30 text-emerald-400 rounded text-[10px] font-bold tracking-wider transition-colors border border-emerald-500/30"
                      >
                        UTILISER
                      </button>
                    )}
                    {item.tradeable && (
                      <button
                        onClick={() => removeItem(slot.slotId)}
                        className="px-3 py-1 bg-red-500/20 hover:bg-red-500/30 text-red-400 rounded text-[10px] font-bold tracking-wider transition-colors border border-red-500/30"
                      >
                        JETER
                      </button>
                    )}
                  </div>
                </div>
              </div>
            )
          })()}

          {/* Player stats bar */}
          <div className="border-t border-zinc-700/30 px-4 py-2">
            <div className="grid grid-cols-3 gap-2 text-[10px]">
              {[
                { label: 'Energie', value: playerStats.energy, color: '#22c55e' },
                { label: 'Faim', value: playerStats.hunger, color: '#f59e0b' },
                { label: 'Humeur', value: playerStats.mood, color: '#8b5cf6' },
              ].map((stat) => (
                <div key={stat.label} className="flex items-center gap-1.5">
                  <span className="text-zinc-500 w-10">{stat.label}</span>
                  <div className="flex-1 h-1.5 bg-zinc-800 rounded-full overflow-hidden">
                    <div
                      className="h-full rounded-full transition-all"
                      style={{ width: `${stat.value}%`, backgroundColor: stat.color }}
                    />
                  </div>
                  <span className="text-zinc-400 w-6 text-right">{stat.value}</span>
                </div>
              ))}
            </div>
          </div>
        </div>
      </div>
    </div>
  )
}

function InventorySlotComponent({
  slot,
  isSelected,
  onSelect,
  onUse,
  onRemove,
  onDragStart,
  onDrop,
}: {
  slot: { slotId: number; item: InventoryItem | null; locked: boolean }
  isSelected: boolean
  onSelect: () => void
  onUse: () => void
  onRemove: () => void
  onDragStart: () => void
  onDrop: () => void
}) {
  const item = slot.item
  const rarityColor = item ? RARITY_COLORS[item.rarity] : '#374151'

  return (
    <div
      draggable={!!item}
      onDragStart={onDragStart}
      onDragOver={(e) => e.preventDefault()}
      onDrop={onDrop}
      onClick={onSelect}
      onDoubleClick={item?.usable ? onUse : undefined}
      className={cn(
        "relative aspect-square rounded-lg border-2 flex flex-col items-center justify-center cursor-pointer transition-all group",
        isSelected
          ? "border-amber-400 bg-amber-400/10 shadow-lg shadow-amber-400/20"
          : slot.locked
            ? "border-zinc-800 bg-zinc-900/50 cursor-not-allowed"
            : item
              ? "border-zinc-700 bg-zinc-800/50 hover:border-zinc-500 hover:bg-zinc-700/30"
              : "border-zinc-800/50 bg-zinc-900/30 hover:border-zinc-700"
      )}
      style={!isSelected && item ? { borderColor: `${rarityColor}40` } : undefined}
    >
      {item ? (
        <>
          <span className="text-lg">{item.icon}</span>
          {item.quantity > 1 && (
            <span className="absolute bottom-0.5 right-1 text-[9px] font-bold text-zinc-300">
              {item.quantity}
            </span>
          )}
          <div
            className="absolute bottom-0 left-0 right-0 h-0.5 rounded-b-lg"
            style={{ backgroundColor: rarityColor }}
          />
          <button
            onClick={(e) => { e.stopPropagation(); onRemove() }}
            className="absolute -top-1 -right-1 w-4 h-4 bg-red-500/80 rounded-full items-center justify-center text-white text-[8px] hidden group-hover:flex"
          >
            <Trash2 className="w-2.5 h-2.5" />
          </button>
        </>
      ) : slot.locked ? (
        <X className="w-3 h-3 text-zinc-700" />
      ) : null}
    </div>
  )
}

export function SpawnParticles({ active }: { active: boolean }) {
  const particlesRef = useRef<THREE.Points>(null)
  const particleCount = 100
  
  const positions = useRef<Float32Array>(
    new Float32Array(particleCount * 3).map((_, i) => {
      const idx = i % 3
      if (idx === 0) return (Math.random() - 0.5) * 3 // x
      if (idx === 1) return Math.random() * 2 + 0.5 // y
      return (Math.random() - 0.5) * 3 // z
    })
  )
  
  useFrame((state) => {
    if (particlesRef.current && active) {
      const positions = particlesRef.current.geometry.attributes.position.array as Float32Array
      
      for (let i = 0; i < particleCount; i++) {
        const i3 = i * 3
        positions[i3 + 1] += 0.01 + Math.sin(state.clock.elapsedTime + i) * 0.005
        
        if (positions[i3 + 1] > 3) {
          positions[i3 + 1] = 0.5
        }
      }
      
      particlesRef.current.geometry.attributes.position.needsUpdate = true
      particlesRef.current.rotation.y += 0.001
    }
  })
  
  if (!active) return null
  
  return (
    <points ref={particlesRef} position={[0, 0, 0]}>
      <bufferGeometry>
        <bufferAttribute
          attach="attributes-position"
          args={[positions.current, 3]}
        />
      </bufferGeometry>
      <pointsMaterial
        size={0.03}
        color="#8b5cf6"
        transparent
        opacity={0.6}
        sizeAttenuation
      />
    </points>
  )
}

export interface DoorControllerState {
  id: string
  roomId: string
  isLocked: boolean
  isAnimating: boolean
  lastInteraction: number
  accessLog: AccessLogEntry[]
}

export interface AccessLogEntry {
  timestamp: number
  cardLevel: string
  granted: boolean
}

const DOOR_OPEN_SPEED = 3.5
const AUTO_CLOSE_DELAY = 5000

export function createDoorController(id: string, roomId: string, isLocked: boolean): DoorControllerState {
  return {
    id,
    roomId,
    isLocked,
    isAnimating: false,
    lastInteraction: 0,
    accessLog: [],
  }
}

export function validateAccess(
  doorState: DoorControllerState,
  cardLevel: number,
  requiredLevel: number
): { granted: boolean; doorState: DoorControllerState } {
  const granted = cardLevel >= requiredLevel
  const logEntry: AccessLogEntry = {
    timestamp: Date.now(),
    cardLevel: cardLevel.toString(),
    granted,
  }

  return {
    granted,
    doorState: {
      ...doorState,
      lastInteraction: Date.now(),
      isAnimating: true,
      accessLog: [...doorState.accessLog.slice(-19), logEntry],
    },
  }
}

export function calculateDoorAngle(
  currentAngle: number,
  targetAngle: number,
  deltaTime: number
): number {
  const diff = targetAngle - currentAngle
  if (Math.abs(diff) < 0.01) return targetAngle
  return currentAngle + diff * Math.min(deltaTime * DOOR_OPEN_SPEED, 1)
}

export function shouldAutoClose(doorState: DoorControllerState): boolean {
  return (
    !doorState.isLocked &&
    !doorState.isAnimating &&
    Date.now() - doorState.lastInteraction > AUTO_CLOSE_DELAY
  )
}

  const { setTheme } = useTheme()

  return (
    <DropdownMenu>
      <DropdownMenuTrigger asChild>
        <Button variant="outline" size="icon">
          <Sun className="h-[1.2rem] w-[1.2rem] scale-100 rotate-0 transition-all dark:scale-0 dark:-rotate-90" />

      </DropdownMenuTrigger>

        </DropdownMenuItem>
        <DropdownMenuItem onClick={() => setTheme("system")}>
          System
        </DropdownMenuItem>
      </DropdownMenuContent>
    </DropdownMenu>
  )
}

import * as React from "react"

type Theme = "dark" | "light" | "system"
type ResolvedTheme = "dark" | "light"

type ThemeProviderProps = {
  children: React.ReactNode
  defaultTheme?: Theme
  storageKey?: string
  disableTransitionOnChange?: boolean
}

type ThemeProviderState = {
  theme: Theme
  setTheme: (theme: Theme) => void
}

const COLOR_SCHEME_QUERY = "(prefers-color-scheme: dark)"
const THEME_VALUES: Theme[] = ["dark", "light", "system"]

const ThemeProviderContext = React.createContext<
  ThemeProviderState | undefined
>(undefined)

function isTheme(value: string | null): value is Theme {
  if (value === null) {
    return false
  }

  return THEME_VALUES.includes(value as Theme)
}

function getSystemTheme(): ResolvedTheme {
  if (window.matchMedia(COLOR_SCHEME_QUERY).matches) {
    return "dark"
  }

  return "light"
}

function disableTransitionsTemporarily() {
  const style = document.createElement("style")
  style.appendChild(
    document.createTextNode(
      "*,*::before,*::after{-webkit-transition:none!important;transition:none!important}"
    )
  )
  document.head.appendChild(style)

  return () => {
    window.getComputedStyle(document.body)
    requestAnimationFrame(() => {
      requestAnimationFrame(() => {
        style.remove()
      })
    })
  }
}

function isEditableTarget(target: EventTarget | null) {
  if (!(target instanceof HTMLElement)) {
    return false
  }

  if (target.isContentEditable) {
    return true
  }

  const editableParent = target.closest(
    "input, textarea, select, [contenteditable='true']"
  )
  if (editableParent) {
    return true
  }

  return false
}

export function ThemeProvider({
  children,
  defaultTheme = "system",
  storageKey = "theme",
  disableTransitionOnChange = true,
  ...props
}: ThemeProviderProps) {
  const [theme, setThemeState] = React.useState<Theme>(() => {
    const storedTheme = localStorage.getItem(storageKey)
    if (isTheme(storedTheme)) {
      return storedTheme
    }

    return defaultTheme
  })

  const setTheme = React.useCallback(
    (nextTheme: Theme) => {
      localStorage.setItem(storageKey, nextTheme)
      setThemeState(nextTheme)
    },
    [storageKey]
  )

  const applyTheme = React.useCallback(
    (nextTheme: Theme) => {
      const root = document.documentElement
      const resolvedTheme =
        nextTheme === "system" ? getSystemTheme() : nextTheme
      const restoreTransitions = disableTransitionOnChange
        ? disableTransitionsTemporarily()
        : null

      root.classList.remove("light", "dark")
      root.classList.add(resolvedTheme)

      if (restoreTransitions) {
        restoreTransitions()
      }
    },
    [disableTransitionOnChange]
  )

  React.useEffect(() => {
    applyTheme(theme)

    if (theme !== "system") {
      return undefined
    }

    const mediaQuery = window.matchMedia(COLOR_SCHEME_QUERY)
    const handleChange = () => {
      applyTheme("system")
    }

    mediaQuery.addEventListener("change", handleChange)

    return () => {
      mediaQuery.removeEventListener("change", handleChange)
    }
  }, [theme, applyTheme])

  React.useEffect(() => {
    const handleKeyDown = (event: KeyboardEvent) => {
      if (event.repeat) {
        return
      }

      if (event.metaKey || event.ctrlKey || event.altKey) {
        return
      }

      if (isEditableTarget(event.target)) {
        return
      }

      if (event.key.toLowerCase() !== "d") {
        return
      }

      setThemeState((currentTheme) => {
        const nextTheme =
          currentTheme === "dark"
            ? "light"
            : currentTheme === "light"
              ? "dark"
              : getSystemTheme() === "dark"
                ? "light"
                : "dark"

        localStorage.setItem(storageKey, nextTheme)
        return nextTheme
      })
    }

    window.addEventListener("keydown", handleKeyDown)

    return () => {
      window.removeEventListener("keydown", handleKeyDown)
    }
  }, [storageKey])

  React.useEffect(() => {
    const handleStorageChange = (event: StorageEvent) => {
      if (event.storageArea !== localStorage) {
        return
      }

      if (event.key !== storageKey) {
        return
      }

      if (isTheme(event.newValue)) {
        setThemeState(event.newValue)
        return
      }

      setThemeState(defaultTheme)
    }

    window.addEventListener("storage", handleStorageChange)

    return () => {
      window.removeEventListener("storage", handleStorageChange)
    }
  }, [defaultTheme, storageKey])

  const value = React.useMemo(
    () => ({
      theme,
      setTheme,
    }),
    [theme, setTheme]
  )

  return (
    <ThemeProviderContext.Provider {...props} value={value}>
      {children}
    </ThemeProviderContext.Provider>
  )
}

export const useTheme = () => {
  const context = React.useContext(ThemeProviderContext)

  if (context === undefined) {
    throw new Error("useTheme must be used within a ThemeProvider")
  }

  return context
}
