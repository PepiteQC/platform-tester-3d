'use client'

import { useState, useRef, useEffect } from 'react'
import { Card } from '@/components/ui/card'
import { Button } from '@/components/ui/button'
import { Textarea } from '@/components/ui/textarea'
import { Spinner } from '@/components/ui/spinner'
import { Send, Brain, Zap } from 'lucide-react'
import { toast } from 'sonner'

interface Message {
  id: string
  role: 'user' | 'assistant'
  content: string
  timestamp: Date
}

export function TroxtChatInterface() {
  const [messages, setMessages] = useState<Message[]>([
    {
      id: '1',
      role: 'assistant',
      content: "I'm TROXT, your proprietary AI intelligence. I can analyze your ideas, understand images, optimize 3D generation, and learn from your preferences. What would you like to create?",
      timestamp: new Date(),
    },
  ])
  const [input, setInput] = useState('')
  const [isLoading, setIsLoading] = useState(false)
  const messagesEndRef = useRef<HTMLDivElement>(null)

  const scrollToBottom = () => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' })
  }

  useEffect(() => {
    scrollToBottom()
  }, [messages])

  const handleSendMessage = async (e: React.FormEvent) => {
    e.preventDefault()
    if (!input.trim()) return

    const userMessage: Message = {
      id: Date.now().toString(),
      role: 'user',
      content: input,
      timestamp: new Date(),
    }

    setMessages(prev => [...prev, userMessage])
    setInput('')
    setIsLoading(true)

    try {
      const response = await fetch('/api/troxt/analyze-prompt', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ prompt: input }),
      })

      const data = await response.json()

      const assistantMessage: Message = {
        id: (Date.now() + 1).toString(),
        role: 'assistant',
        content: data.analysis || "I've processed your request. Let me analyze and optimize the generation strategy.",
        timestamp: new Date(),
      }

      setMessages(prev => [...prev, assistantMessage])
    } catch (error) {
      toast.error('TROXT analysis failed')
      console.error('[v0] TROXT error:', error)
    } finally {
      setIsLoading(false)
    }
  }

  return (
    <Card className="bg-card border-primary/20 flex flex-col h-full">
      <div className="p-4 border-b border-primary/10 flex items-center gap-2">
        <Brain className="size-4 text-primary" />
        <div>
          <h3 className="text-sm font-semibold text-foreground">TROXT Agent</h3>
          <p className="text-xs text-muted-foreground">Your AI Intelligence</p>
        </div>
      </div>

      <div className="flex-1 overflow-y-auto p-4 space-y-4">
        {messages.map(msg => (
          <div key={msg.id} className={`flex ${msg.role === 'user' ? 'justify-end' : 'justify-start'}`}>
            <div
              className={`max-w-xs lg:max-w-md px-3 py-2 rounded-lg text-sm ${
                msg.role === 'user'
                  ? 'bg-primary text-primary-foreground'
                  : 'bg-muted text-muted-foreground border border-border'
              }`}
            >
              {msg.content}
            </div>
          </div>
        ))}
        {isLoading && (
          <div className="flex justify-start">
            <div className="bg-muted border border-border px-3 py-2 rounded-lg">
              <Spinner className="size-4" />
            </div>
          </div>
        )}
        <div ref={messagesEndRef} />
      </div>

      <form onSubmit={handleSendMessage} className="p-4 border-t border-primary/10 space-y-2">
        <Textarea
          placeholder="Ask TROXT anything... or describe what you want to create"
          value={input}
          onChange={e => setInput(e.target.value)}
          className="resize-none h-16 text-sm"
          disabled={isLoading}
        />
        <Button type="submit" size="sm" className="w-full gap-2" disabled={isLoading || !input.trim()}>
          <Send className="size-3" />
          Send to TROXT
        </Button>
      </form>
    </Card>
  )
}
